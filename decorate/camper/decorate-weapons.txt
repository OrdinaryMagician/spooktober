Actor DeepSMG : DoomWeapon 12121
{
	//$Category SpookyWeapon
	Weapon.AmmoGive 12
	Weapon.SelectionOrder 1901
	Weapon.AmmoUse 1
	Weapon.AmmoType "Clip"
	Weapon.SlotNumber 2
	Obituary "$OB_MPPISTOL"
	Inventory.Pickupmessage "Picked up smg"
	Tag "$TAG_PISTOL"
	+CHEATNOTWEAPON
	States
	{
		Spawn:
			SSMG D -1
			Stop
		Ready:
			SSMG A 1 A_WeaponReady
			Loop
		Deselect:
			SSMG A 1 A_Lower
			Loop
		Select:
			SSMG A 1 A_Raise
			Loop
		Fire:
			SSMG A 2 A_PlaySound("weapons/chngun", CHAN_WEAPON)
			SSMG B 1 A_FireBullets (5.6, 0, 1, 15, "BulletPuff")
			SSMG C 1 A_GunFlash
			SSMG B 0 A_ReFire
			Goto Ready
	}
}
