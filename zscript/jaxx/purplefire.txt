Class Ember1purple : Flame1Apurple
{
	default
	{
	Scale 0.1;
	}
	States
	{
	Spawn:
		WMBR A 0;
		WMBR A 0 {
	A_ChangeVelocity(frandom(-1,1),frandom(-1,1),frandom(4,16));
	A_SetRoll(random(0,90));
	}
		WMBR A 32 Bright;
	Death:
		WMBR A 2 Bright A_FadeOut(0.1);
		Loop;
	}
}
