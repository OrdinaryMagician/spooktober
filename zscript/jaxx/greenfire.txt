Class Ember1green : Flame1Agreen
{
	default
	{
	Scale 0.1;
	}
	States
	{
	Spawn:
		GMBR A 0;
		GMBR A 0 {
	A_ChangeVelocity(frandom(-1,1),frandom(-1,1),frandom(4,16));
	A_SetRoll(random(0,90));
	}
		GMBR A 32 Bright;
	Death:
		GMBR A 2 Bright A_FadeOut(0.1);
		Loop;
	}
}
